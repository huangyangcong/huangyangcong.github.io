    再执行sudo mysqld_safe –user=mysql –skip-grant-tables
    mysql -u root mysql
    ALTER user ‘root’@’localhost’ IDENTIFIED BY ‘newpassward’;
    FLUSH PRIVILEGES ./bin/mysqld_safe –defaults-file=conf/3307my.cnf –skip-grant-tables&
    mysql –socket=/home/hyc/software/mysql-8.0.18-linux-glibc2.12-x86_64/data_3307/3307_mysql.sock -uroot -P3307 -p
    use mysql
    flush privileges
    ALTER USER ‘root’@’localhost’ IDENTIFIED BY ‘123456’; #mysql5.7.6
    update mysql.user set authentication_string=password(‘new_password’) where user=’root’ and Host =’localhost’;
    或者
    update user set password = password(‘new-password’) where user = ‘root’ ;
    flush privileges;

#mysql5.7.9
ALTER USER ‘jeffrey’@’localhost’ IDENTIFIED WITH mysql_native_password AS ‘authentication_string字段内容’;
或者
通过 SET GLOBAL validate_password_policy=’LOW’; 命令，降低安全等级后，就可以直接使用，限制是必须8个字符以上；
ALTER USER USER() IDENTIFIED BY ‘news_password’;

#mysql8
use mysql;
flush privileges;
ALTER USER ‘root’@’localhost’ IDENTIFIED BY ‘MyNewPass’;

default_password_lifetime=0
